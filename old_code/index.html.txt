  function parseAndFormatDate(dateInput) {
    if (!dateInput) {
      return 'N/A';
 }
    let date;
    const directDate = new Date(dateInput);
 if (!isNaN(directDate.getTime())) {
      date = directDate;
 } else if (typeof dateInput === 'string' && dateInput.includes('/')) {
      const parts = dateInput.split('/');
 if (parts.length === 3) {
        const day = parseInt(parts[0], 10);
 const month = parseInt(parts[1], 10);
        const year = parseInt(parts[2], 10);
 if (year > 1900 && month >= 1 && month <= 12 && day >= 1 && day <= 31) {
          date = new Date(year, month - 1, day);
 }
      }
    }
    if (date && !isNaN(date.getTime())) {
      const day = date.getDate();
 const month = date.getMonth() + 1;
      const buddhistYear = date.getFullYear() + 543;
      return `${day}/${month}/${buddhistYear}`;
    }
    return 'N/A';
 }

async function loadSystemSettings() {
    try {
        // 1. ลองดึงข้อมูลจาก Cache ในเครื่องผู้ใช้ก่อน (Local Storage)
        const cachedSettings = localStorage.getItem('systemSettingsCache');
        if (cachedSettings) {
            const settings = JSON.parse(cachedSettings);
            if (settings.logo) applyLogo(settings.logo);
            if (settings.background) applyLoginBackground(settings.background);
            console.log("Loaded settings from cache (Fast Load)");
        }

        // 2. โหลดข้อมูลล่าสุดจาก Server (ทำอยู่เบื้องหลัง) เพื่ออัปเดตถ้ามีการเปลี่ยนแปลง
        const serverSettings = await new Promise((resolve, reject) => {
            google.script.run
                .withSuccessHandler(resolve)
                .withFailureHandler(reject)
                .getSystemSettingsBundle(); // เราจะสร้างฟังก์ชันนี้ใน code.gs เพิ่มเติมเพื่อลดการเรียกหลายรอบ
        });

        if (serverSettings) {
            if (serverSettings.logo) applyLogo(serverSettings.logo);
            if (serverSettings.background) applyLoginBackground(serverSettings.background);

            // บันทึกลง Cache ไว้ใช้ครั้งหน้า
            localStorage.setItem('systemSettingsCache', JSON.stringify({
                logo: serverSettings.logo,
                background: serverSettings.background
            }));
        }

    } catch (error) {
        console.error("Failed to load system settings:", error);
        // ถ้าโหลดไม่สำเร็จ และไม่มี Cache ให้ใช้ค่า Default
        if (!localStorage.getItem('systemSettingsCache')) {
            applyLogo(null);
            applyLoginBackground(null);
        }
    }
}

function updateAndPopulateTimeDropdowns() {
  const selectedDateValue = otRequestDateInput.value;
  // หากยังไม่ได้เลือกวันที่ ให้ล้างค่าตัวเลือกเวลา
  if (!selectedDateValue) {
    otOverTimeStartInput.innerHTML = '';
    otOverTimeEndInput.innerHTML = '';
    calculateTotalHours();
 // คำนวณชั่วโมงใหม่ (ซึ่งจะเป็น 0)
    return;
  }

  // สร้าง Object วันที่และดึงหมายเลขวันในสัปดาห์ (0 = อาทิตย์, 6 = เสาร์)
  const selectedDate = new Date(selectedDateValue + 'T00:00:00');
 // 'T00:00:00' เพื่อให้แน่ใจว่าได้เวลาท้องถิ่นที่ถูกต้อง
  const dayOfWeek = selectedDate.getDay();
  const isHoliday = publicHolidays.includes(selectedDateValue);
 if (dayOfWeek === 6 || dayOfWeek === 0 || isHoliday) {
  // ถ้าเป็นวันหยุด (ส-อา) หรือวันหยุดราชการ ให้ใช้เวลา 08:30 - 16:30
  generateTimeOptions(8, 30, 16, 30);
 } else {
  // ถ้าเป็นวันธรรมดา (จ-ศ) ให้ใช้เวลา 15:30 - 20:30
  generateTimeOptions(15, 30, 20, 30);
 }
}

/**
 * [โค้ดอัปเดต] ฟังก์ชันช่วยสำหรับสร้างตัวเลือกเวลาตามช่วงที่กำหนด
 * @param {number} startHour ชั่วโมงเริ่มต้น
 * @param {number} startMinute นาทีเริ่มต้น
 * @param {number} endHour ชั่วโมงสิ้นสุด
 * @param {number} endMinute นาทีสิ้นสุด
 */
function generateTimeOptions(startHour, startMinute, endHour, endMinute) {
  // ล้างค่าตัวเลือกเก่า
  otOverTimeStartInput.innerHTML = '';
 otOverTimeEndInput.innerHTML = '';

  // สร้างตัวเลือกเวลาทุกๆ 30 นาที
  for (let h = startHour; h <= endHour; h++) {
    for (let m = 0; m < 60; m += 30) {
      // ตรวจสอบว่าเวลาที่สร้างอยู่ในช่วงที่กำหนดหรือไม่
      const currentTimeInMinutes = h * 60 + m;
 const startTimeInMinutes = startHour * 60 + startMinute;
      const endTimeInMinutes = endHour * 60 + endMinute;
 if (currentTimeInMinutes >= startTimeInMinutes && currentTimeInMinutes <= endTimeInMinutes) {
        const time = `${String(h).padStart(2, '0')}:${String(m).padStart(2, '0')}`;
 const optionStart = document.createElement('option');
        optionStart.value = time;
        optionStart.textContent = time;
        otOverTimeStartInput.appendChild(optionStart);

        const optionEnd = document.createElement('option');
        optionEnd.value = time;
 optionEnd.textContent = time;
        otOverTimeEndInput.appendChild(optionEnd);
      }
    }
  }

  // [จุดที่เพิ่ม] กำหนดให้เวลาสิ้นสุดเลือกค่าสุดท้าย (เวลาสูงสุด) เป็นค่าเริ่มต้น
  if (otOverTimeEndInput.options.length > 0) {
      otOverTimeEndInput.selectedIndex = otOverTimeEndInput.options.length - 1;
 }
  
  // หลังจากสร้างตัวเลือกและกำหนดค่าเริ่มต้นเสร็จ ให้คำนวณชั่วโมงใหม่
  calculateTotalHours();
}
function toThaiNumerals(input) {
  if (input === null || typeof input === 'undefined') return '';
 const str = String(input);
  const thaiNumerals = ['๐', '๑', '๒', '๓', '๔', '๕', '๖', '๗', '๘', '๙'];
 return str.replace(/[0-9]/g, (d) => thaiNumerals[parseInt(d, 10)]);
}
function formatThaiDate(dateString) {
  if (!dateString) return 'N/A';
  const date = new Date(dateString);
 if (isNaN(date.getTime())) return 'N/A';
  const options = {
    year: 'numeric',
    month: 'long',
    day: 'numeric',
    timeZone: 'Asia/Bangkok'
  };
 const thaiDate = new Intl.DateTimeFormat('th-TH-u-nu-thai', options).format(date);
  return thaiDate;
}
  function calculateTotalHours() {
    const start = otOverTimeStartInput.value;
 const end = otOverTimeEndInput.value;
if (start && end) {
      const startTime = new Date(`2000/01/01 ${start}`);
 const endTime = new Date(`2000/01/01 ${end}`);
      if (endTime < startTime) {
        endTime.setDate(endTime.getDate() + 1);
 }
      const diffMs = endTime - startTime;
 const hours = diffMs / (1000 * 60 * 60);
      otTotalHoursInput.value = hours.toFixed(2);
 } else {
      otTotalHoursInput.value = '';
 }
  }

async function handleGenerateMemo() {
    const generateBtn = document.getElementById('memo-generate-button');
    generateBtn.disabled = true;
    generateBtn.textContent = 'กำลังสร้าง...';
    
    const selectedRows = document.querySelectorAll('#memo-table-body .memo-row-select:checked');
    if (selectedRows.length === 0) {
        showMessage("กรุณาเลือกรายการที่ต้องการสร้างบันทึก", true);
        generateBtn.disabled = false;
        generateBtn.textContent = 'ร่างบันทึก';
        return;
    }
    
    const selectedData = [];
    selectedRows.forEach(row => {
        const tr = row.closest('tr');
        selectedData.push({
            requestDate: tr.dataset.rawDate, 
            fullName: tr.dataset.fullName,
            position: tr.dataset.position,
            reason: tr.dataset.reason
        });
    });
    
    const format = document.querySelector('input[name="memo-format"]:checked').value;
    
    // รับค่า Controller Type (ป้องกัน Error กรณีหน้าจอไม่มีปุ่มเลือก)
    let controllerType = 'auto';
    const controllerRadio = document.querySelector('input[name="memo-controller"]:checked');
    if (controllerRadio) {
        controllerType = controllerRadio.value;
    }

    try {
        const result = await new Promise((resolve, reject) => {
            google.script.run
                .withSuccessHandler(resolve)
                .withFailureHandler(reject)
                .generateMemoDocument(selectedData, format, controllerType);
        });

            const linkContainer = document.getElementById('memo-link-container');
            const linkDisplay = document.getElementById('memo-link-display');

          .withSuccessHandler(resolve)
          .withFailureHandler(reject)
          .doLogin(usernameInput.value, passwordInput.value);
      });

      if (response.success) {
        isLoggedIn = true;
        userRole = response.role;
        userId = response.userId;
        const fullName = `${response.firstName} ${response.lastName}`.trim();
        userName = fullName;
        userPosition = response.position || '';

        const userSession = {
          isLoggedIn: true,
          role: userRole,
          userId: userId,
          userName: userName,
          userPosition: userPosition
        };
        
        // --- จุดแก้ไขสำคัญ: ใช้ localStorage แทน sessionStorage เพื่อให้จำค่าแม้ปิดบราวเซอร์ ---
        localStorage.setItem('loggedInUser', JSON.stringify(userSession)); 
        
        await fetchPublicHolidays();
        showMessage('เข้าสู่ระบบสำเร็จ!');
        renderScreen();
      } else {
        loginMessage.textContent = `เข้าสู่ระบบไม่สำเร็จ: ${response.message}`;
        showMessage(`เข้าสู่ระบบไม่สำเร็จ: ${response.message}`, true);
      }
    } catch (error) {
      console.error('Login error:', error);
      loginMessage.textContent = 'เกิดข้อผิดพลาดในการเข้าสู่ระบบ. กรุณาลองใหม่อีกครั้ง';
      showMessage('เกิดข้อผิดพลาดในการเข้าสู่ระบบ. กรุณาลองใหม่อีกครั้ง', true);
    } finally {

      loginButton.textContent = 'เข้าสู่ระบบ';
    }
  }

async function handleCancelRequest(event) {
    const button = event.target;
    const requestId = button.dataset.requestId;
 const confirmed = await showConfirmationModal(`คุณแน่ใจหรือไม่ที่จะยกเลิกคำขอ ${requestId}?`);
    if (!confirmed) {
        return;
 }
    button.disabled = true;
    button.textContent = 'กำลังยกเลิก...';
 try {
        const response = await new Promise((resolve, reject) => {
            google.script.run
                .withSuccessHandler(resolve)
                .withFailureHandler(reject)
                .cancelOTRequest(requestId, userId);
        });
 if (response.success) {
            showMessage(response.message);
 // โหลดข้อมูลใหม่ตาม role ของผู้ใช้
            if (userRole === 'employee') {
                fetchEmployeeOTRequests();
 } else if (userRole === 'supervisor') {
                fetchSupervisorOwnRequests();
 }
        } else {
            showMessage(response.message, true);
 button.disabled = false;
            button.textContent = 'ยกเลิกคำขอ';
        }
    } catch (error) {
        console.error('Error in handleCancelRequest:', error);
 showMessage('เกิดข้อผิดพลาดในการเชื่อมต่อเพื่อยกเลิกคำขอ', true);
        button.disabled = false;
        button.textContent = 'ยกเลิกคำขอ';
    }
}

function renderRequestRows(requests, tableBodyEl) {
  tableBodyEl.innerHTML = '';
  requests.forEach(req => {
    const row = tableBodyEl.insertRow();
    let statusClass = 'bg-gray-100 text-gray-800';
    if (req.ApprovalStatus.includes('Pending') || req.ApprovalStatus.includes('Supervisor')) {
       statusClass = 'bg-yellow-100 text-yellow-800';
    } else if (req.ApprovalStatus.includes('Rejected') || req.ApprovalStatus.includes('ยกเลิก')) {
       statusClass = 'bg-red-100 text-red-800';
    } else if (req.ApprovalStatus.includes('Approved by Commander')) {
       statusClass = 'bg-green-100 text-green-800';
    }

    const canCancel = (userRole === 'employee' && req.ApprovalStatus === 'Pending Supervisor Approval') ||
                      (userRole === 'supervisor' && (req.ApprovalStatus === 'Pending Supervisor Approval' || req.ApprovalStatus === 'Approved by Supervisor'));

    const cancelButtonHtml = canCancel
      ? `<button data-request-id="${req.RequestID}" class="cancel-request-button bg-gray-500 text-white px-3 py-1 rounded-lg text-xs hover:bg-gray-600 transition duration-200">ยกเลิกคำขอ</button>`
      : '-';
    row.innerHTML = `
      <td class="px-4 py-3 whitespace-nowrap text-sm text-gray-800">${req.RequestID}</td>
      <td class="px-4 py-3 whitespace-nowrap text-sm text-gray-800">${req.RequestDate ? new Date(req.RequestDate).toLocaleDateString('th-TH') : 'N/A'}</td>
      <td class="px-4 py-3 whitespace-nowrap text-sm text-gray-800">${req.OverTimeStart} - ${req.OverTimeEnd} (${req.TotalHours} ชม.)</td>
      <td class="px-4 py-3 text-sm text-gray-800" title="${req.Reason}">${req.Reason}</td>
      <td class="px-4 py-3 whitespace-nowrap"><span class="px-2 inline-flex text-xs leading-5 font-semibold rounded-full ${statusClass}">${translateStatus(req.ApprovalStatus)}</span></td>
      <td class="px-4 py-3 whitespace-nowrap text-center">${cancelButtonHtml}</td>
    `;
  });

  document.querySelectorAll('.cancel-request-button').forEach(button => {
    button.addEventListener('click', handleCancelRequest);
  });
}
function setActiveAdminTab(tabName) {
  for (const key in adminTabs) {
    if (key === tabName) {
      adminTabs[key].classList.add('text-blue-600', 'border-blue-600');
      adminTabs[key].classList.remove('text-gray-500', 'border-transparent', 'hover:text-gray-600', 'hover:border-gray-300');
      adminTabPanes[key].classList.remove('hidden');
    } else {
      adminTabs[key].classList.remove('text-blue-600', 'border-blue-600');
      adminTabs[key].classList.add('text-gray-500', 'border-transparent', 'hover:text-gray-600', 'hover:border-gray-300');
      adminTabPanes[key].classList.add('hidden');
    }
  }
  if (tabName === 'user-management') {
    fetchUsers();
  } else if (tabName === 'ot-report') {
    fetchDepartmentsForReport();
    fetchOTReport();
  }
}

async function handleChangePasswordSubmit(e) {
  e.preventDefault();
  passwordErrorMessage.classList.add('hidden');
  passwordErrorMessage.textContent = '';
  changePasswordSubmitButton.disabled = true;
  changePasswordSubmitButton.textContent = 'กำลังดำเนินการ...';
  const currentPassword = document.getElementById('current-password-input').value;
  const newPassword = document.getElementById('new-password-input').value;
  const confirmPassword = document.getElementById('confirm-password-input').value;
  if (newPassword.length < 4) {
    passwordErrorMessage.textContent = 'รหัสผ่านใหม่ต้องมีความยาวอย่างน้อย 4 ตัวอักษร';

    changePasswordSubmitButton.disabled = false;
    changePasswordSubmitButton.textContent = 'ยืนยันการเปลี่ยนรหัสผ่าน';
    return;
  }
  if (newPassword !== confirmPassword) {
    passwordErrorMessage.textContent = 'รหัสผ่านใหม่และการยืนยันไม่ตรงกัน';
    passwordErrorMessage.classList.remove('hidden');
    changePasswordSubmitButton.disabled = false;
    changePasswordSubmitButton.textContent = 'ยืนยันการเปลี่ยนรหัสผ่าน';
    return;
  }
  try {
    const response = await new Promise((resolve, reject) => {
      google.script.run
        .withSuccessHandler(resolve)
        .withFailureHandler(reject)
        .changePassword(userId, currentPassword, newPassword);
    });
    if (response.success) {
      showMessage(response.message);
      changePasswordModal.classList.add('hidden');
      changePasswordForm.reset();
    } else {
      passwordErrorMessage.textContent = response.message;
      passwordErrorMessage.classList.remove('hidden');

    passwordErrorMessage.textContent = `เกิดข้อผิดพลาด: ${error.message}`;

function handleReportSearch() {
  const searchTerm = reportSearchInput.value.toLowerCase().trim();
  const rows = otReportTableBody.getElementsByTagName('tr');
  for (const row of rows) {
    const rowText = row.textContent.toLowerCase();
    if (rowText.includes(searchTerm)) {
      row.style.display = '';
    } else {
      row.style.display = 'none';
    }
  }
}

async function handleExportClick(sheetName, buttonId, messageId) {
  const btn = document.getElementById(buttonId);
  const msgArea = document.getElementById(messageId);
  btn.disabled = true;
  btn.textContent = 'กำลังส่งออก...';
  msgArea.textContent = 'กำลังเตรียมข้อมูล...';

    const result = await new Promise((resolve, reject) => {
      google.script.run
        .withSuccessHandler(resolve)
        .withFailureHandler(reject)
        .handleExport(sheetName);
    });
    const blob = new Blob(["\uFEFF" + result.csvContent], { type: 'text/csv;charset=utf-8;' });
    const link = document.createElement("a");
    if (link.download !== undefined) {
      const url = URL.createObjectURL(blob);
      link.setAttribute("href", url);
      link.setAttribute("download", result.fileName);
      link.style.visibility = 'hidden';
      document.body.appendChild(link);
      link.click();
      document.body.removeChild(link);
    }
    msgArea.textContent = 'ส่งออกข้อมูลสำเร็จ!';
  } catch (error) {
    console.error('Export Error:', error);
    msgArea.textContent = `เกิดข้อผิดพลาด: ${error.message}`;
  } finally {
    btn.disabled = false;
    btn.textContent = 'ส่งออกข้อมูล (CSV)';
    setTimeout(() => { msgArea.textContent = ''; }, 5000);
  }
}

/**
 * จัดการเมื่อมีการเลือกไฟล์ CSV เพื่อนำเข้า
 */
function handleImportFile(event, sheetName, messageId) {
  const file = event.target.files[0];
  if (!file) return;

  const msgArea = document.getElementById(messageId);
  msgArea.textContent = `กำลังอ่านไฟล์ ${file.name}...`;
  const btn = event.target.closest('label');
  if (btn) btn.classList.add('opacity-50', 'cursor-not-allowed');
  const reader = new FileReader();
  reader.onload = async (e) => {
    const csvContent = e.target.result;
    msgArea.textContent = 'กำลังนำเข้าข้อมูลสู่ระบบ...';
    try {
      const result = await new Promise((resolve, reject) => {
        google.script.run
          .withSuccessHandler(resolve)
          .withFailureHandler(reject)
          .handleImportCsv(csvContent, sheetName);
      });
      if (result.success) {
        msgArea.textContent = result.message;
        if(sheetName === 'tb_User') fetchUsers(); // หากเป็นชีต User ให้โหลดข้อมูลใหม่
      } else {
        msgArea.textContent = `นำเข้าไม่สำเร็จ: ${result.message}`;
      }

    } catch (error) {
      console.error('Import Error:', error);
      msgArea.textContent = `เกิดข้อผิดพลาดรุนแรง: ${error.message}`;
    } finally {
        if (btn) btn.classList.remove('opacity-50', 'cursor-not-allowed');
        event.target.value = ''; // เคลียร์ค่าไฟล์ input
        setTimeout(() => { msgArea.textContent = ''; }, 8000);
      }
  };
  reader.readAsText(file, 'UTF-8'); // อ่านไฟล์เป็น UTF-8
}

// 3. Time & Hours
function generateTimeOptionsForBulk(sh, sm, eh, em) {
    const startSelect = document.getElementById('bulk-time-start');
    const endSelect = document.getElementById('bulk-time-end');
    startSelect.innerHTML = ''; endSelect.innerHTML = '';
    for (let h = 0; h < 24; h++) {
        for (let m = 0; m < 60; m += 30) {
             if (h >= 6) {
                 const t = `${String(h).padStart(2,'0')}:${String(m).padStart(2,'0')}`;
                 startSelect.add(new Option(t, t)); endSelect.add(new Option(t, t));
             }
        }
    }
    startSelect.value = "16:30"; endSelect.value = "20:30";
    calculateBulkHours();
    startSelect.onchange = calculateBulkHours; endSelect.onchange = calculateBulkHours;
}

function calculateBulkHours() {
    const s = document.getElementById('bulk-time-start').value;
    const e = document.getElementById('bulk-time-end').value;
    if (s && e) {
        let sd = new Date(`2000/01/01 ${s}`), ed = new Date(`2000/01/01 ${e}`);
        if (ed < sd) ed.setDate(ed.getDate() + 1);
        document.getElementById('bulk-total-hours').value = ((ed - sd) / 36e5).toFixed(2);
    }
}

function handleAddToQueue() {
    const dateInput = document.getElementById('bulk-dates');
    // ดึงค่าวันที่และกรองช่องว่างทิ้ง
    const dates = dateInput.value.split(',').filter(d => d.trim() !== '');
    const users = document.querySelectorAll('.bulk-user-checkbox:checked');
    const task = document.getElementById('bulk-task-select').value;
    
    // ค่าเวลา
    const ts = document.getElementById('bulk-time-start').value;
    const te = document.getElementById('bulk-time-end').value;
    const th = document.getElementById('bulk-total-hours').value;

    // Validation Check
    if (dates.length === 0) return alert("กรุณาเลือกวันที่");
    if (users.length === 0) return alert("กรุณาเลือกพนักงาน");
    if (!task) return alert("กรุณาเลือกภารกิจ");

    // วนลูปเพิ่มข้อมูลเข้า Queue
    dates.forEach(d => {
        users.forEach(u => {
            bulkQueue.push({
                userId: u.value,
                fullName: u.dataset.fullname,
                requestDate: d.trim(),
                overTimeStart: ts,
                overTimeEnd: te,
                totalHours: th,
                reason: task
            });
        });
    });

    renderBulkQueue(); // อัปเดตตารางรายการด้านล่าง

    // ============================================
    // [จุดที่เพิ่ม] รีเซ็ตค่าเพื่อป้องกัน Human Error
    // ============================================
    
    // 1. ล้างรายชื่อคนที่เลือก (เพื่อไม่ให้เผลอให้คนเดิมซ้ำ ถ้าไม่ได้ตั้งใจ)
    document.querySelectorAll('.bulk-user-checkbox').forEach(cb => cb.checked = false);
    updateBulkSelectedCount();
    const selectAllCheck = document.getElementById('bulk-select-all');
    if(selectAllCheck) selectAllCheck.checked = false;

    // 2. ล้างวันที่ที่เลือกไว้ (สำคัญ! แก้ปัญหาเผลอติดวันเก่ามา)
    if (dateInput._flatpickr) {
        dateInput._flatpickr.clear(); // สั่งเคลียร์ปฏิทิน Flatpickr
    } else {
        dateInput.value = ""; // เคลียร์ค่าแบบปกติ (Fallback)
    }

    // 3. (ทางเลือก) ล้างภารกิจด้วยไหม? 
    // ปกติมักจะทำภารกิจเดิมต่อ ถ้าไม่อยากให้ล้างภารกิจ ก็ไม่ต้องเพิ่มโค้ดล้าง task
    // document.getElementById('bulk-task-select').value = ""; 

    alert("เพิ่มรายการลงคิวเรียบร้อยแล้ว กรุณาเลือกวันที่ใหม่สำหรับรายการต่อไป");
}
function renderBulkQueue() {
    const tbody = document.getElementById('bulk-queue-table-body');
    tbody.innerHTML = '';
    const reversed = [...bulkQueue].reverse(); // โชว์รายการล่าสุดบนสุด
    
    if (reversed.length === 0) tbody.innerHTML = '<tr><td colspan="6" class="text-center py-4 text-gray-400">ว่างเปล่า</td></tr>';
    
    reversed.forEach((item, idx) => {
        const realIdx = bulkQueue.length - 1 - idx;
        const tr = document.createElement('tr');
        tr.innerHTML = `
            <td class="px-4 py-2 text-sm">${realIdx + 1}</td>
            <td class="px-4 py-2 text-sm">${item.requestDate}</td>
            <td class="px-4 py-2 text-sm text-gray-500">${item.overTimeStart}-${item.overTimeEnd}</td>
            <td class="px-4 py-2 text-sm font-bold">${item.fullName}</td>
            <td class="px-4 py-2 text-sm text-blue-600">${item.reason}</td>
            <td class="px-4 py-2 text-center"><button onclick="removeFromQueue(${realIdx})" class="text-red-500 font-bold">x</button></td>
        `;
        tbody.appendChild(tr);
    });
    
    document.getElementById('bulk-queue-count').innerText = `${bulkQueue.length} รายการ`;
    document.getElementById('bulk-save-all-btn').disabled = bulkQueue.length === 0;
}

function removeFromQueue(idx) {
    bulkQueue.splice(idx, 1);
    renderBulkQueue();
}

    finally { btn.disabled = false; btn.innerText = "ยืนยันบันทึกข้อมูลทั้งหมด"; }
}

           }).withFailureHandler((err) => {

      google.script.run.withSuccessHandler(res => appendMessage(res, 'ai')).withFailureHandler(err => appendMessage({text:'Error'}, 'ai')).askGeminiAboutOtData(q);
  });

  const loggedInUserJSON = localStorage.getItem('loggedInUser');
  if (loggedInUserJSON) {

      const userSession = JSON.parse(loggedInUserJSON);
      if (userSession.isLoggedIn) {

        userRole = userSession.role;
        userId = userSession.userId;

        renderScreen();
      }
    } catch (e) {
      localStorage.removeItem('loggedInUser');
      renderScreen();

  } else {

  }

      locale: "th"
  });
